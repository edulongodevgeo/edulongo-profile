2. Quais times têm a maior média de gols por partida? * baseado em suas participações, e não no total do recorte histórico.

SELECT
    times.clube,
    AVG(gols.mandante_placar + gols.visitante_placar) AS media_gols_por_partida
FROM brasileiro_pontos_corridos.campeonato_brasileiro_full AS gols
JOIN (
    SELECT
        clube,
        COUNT(DISTINCT id) AS jogos_disputados
    FROM (
        SELECT
            id,
            mandante AS clube
        FROM brasileiro_pontos_corridos.campeonato_brasileiro_full
        UNION ALL
        SELECT
            id,
            visitante AS clube
        FROM brasileiro_pontos_corridos.campeonato_brasileiro_full
    ) AS jogos
    GROUP BY clube
) AS times ON gols.mandante = times.clube OR gols.visitante = times.clube
GROUP BY times.clube
ORDER BY media_gols_por_partida DESC;

Quais jogadores têm a maior quantidade de chutes/chutes no alvo no campeonato?

SELECT
    g.atleta,
    SUM(e.chutes) AS total_chutes,
    SUM(e.chutes_no_alvo) AS total_chutes_no_alvo
FROM brasileiro_pontos_corridos.campeonato_brasileiro_gols AS g
JOIN brasileiro_pontos_corridos.campeonato_brasileiro_estatisticas_full AS e
    ON g.partida_id = e.partida_id AND g.clube = e.clube
GROUP BY g.atleta
ORDER BY total_chutes DESC, total_chutes_no_alvo DESC
LIMIT 20;

Qual é a quantidade média de cartões amarelos e vermelhos por rodada do campeonato brasileiro (filtrado por ANO)?

Quais partidas têm a maior quantidade de cartões (+ ano)?

SELECT
    cbf.id AS partida_id,
    cbf.rodata,
    EXTRACT(YEAR FROM cbf.data) AS ano,
    cbf.mandante AS time_mandante,
    cbf.visitante AS time_visitante,
    SUM(CASE WHEN cartao = 'Amarelo' THEN 1 ELSE 0 END) AS quantidade_cartoes_amarelos,
    SUM(CASE WHEN cartao = 'Vermelho' THEN 1 ELSE 0 END) AS quantidade_cartoes_vermelhos
FROM brasileiro_pontos_corridos.campeonato_brasileiro_full cbf
JOIN brasileiro_pontos_corridos.campeonato_brasileiro_cartoes cbc
    ON cbf.id = cbc.partida_id
GROUP BY cbf.id, cbf.rodata, ano, time_mandante, time_visitante
ORDER BY quantidade_cartoes_vermelhos DESC;

Quais times têm a maior quantidade de cartões amarelos e vermelhos por partida e ano?

SELECT
    cbf.id AS partida_id,
    cbf.rodata,
    EXTRACT(YEAR FROM cbf.data) AS ano,
    cbc.clube AS time,
    SUM(CASE WHEN cartao = 'Amarelo' THEN 1 ELSE 0 END) AS quantidade_cartoes_amarelos,
    SUM(CASE WHEN cartao = 'Vermelho' THEN 1 ELSE 0 END) AS quantidade_cartoes_vermelhos
FROM brasileiro_pontos_corridos.campeonato_brasileiro_full cbf
JOIN brasileiro_pontos_corridos.campeonato_brasileiro_cartoes cbc
    ON cbf.id = cbc.partida_id
GROUP BY cbf.id, cbf.rodata, ano, time
ORDER BY quantidade_cartoes_amarelos DESC, quantidade_cartoes_vermelhos DESC;

Quais jogadores têm a maior quantidade de cartões amarelos e vermelhos no campeonato?

SELECT
    atleta,
    COUNT(*) FILTER (WHERE cartao = 'Amarelo') AS total_cartoes_amarelos,
    COUNT(*) FILTER (WHERE cartao = 'Vermelho') AS total_cartoes_vermelhos
FROM brasileiro_pontos_corridos.campeonato_brasileiro_cartoes

ORDER BY total_cartoes_vermelhos DESC, total_cartoes_amarelos DESC;

Qual é a quantidade média de impedimentos por partida do campeonato brasileiro?

SELECT
    AVG(estatisticas.impedimentos) AS media_impedimentos_por_partida
FROM brasileiro_pontos_corridos.campeonato_brasileiro_estatisticas_full AS estatisticas
JOIN brasileiro_pontos_corridos.campeonato_brasileiro_full AS jogos ON estatisticas.partida_id = jogos.id;

Quais times têm a maior quantidade de impedimentos por partida (+ ano)?

SELECT
    ano,
    time,
    SUM(quantidade_de_impedimentos) AS total_impedimentos
FROM
    (
        SELECT
            cbf.data AS ano,
            cbf.mandante AS time,
            cbe.impedimentos AS quantidade_de_impedimentos
        FROM brasileiro_pontos_corridos.campeonato_brasileiro_estatisticas_full cbe
        INNER JOIN brasileiro_pontos_corridos.campeonato_brasileiro_full cbf
            ON cbe.partida_id = cbf.id

        UNION ALL

        SELECT
            cbf.data AS ano,
            cbf.visitante AS time,
            cbe.impedimentos AS quantidade_de_impedimentos
        FROM brasileiro_pontos_corridos.campeonato_brasileiro_estatisticas_full cbe
        INNER JOIN brasileiro_pontos_corridos.campeonato_brasileiro_full cbf
            ON cbe.partida_id = cbf.id
    ) AS total_impedimentos_por_time
GROUP BY ano, time
ORDER BY total_impedimentos DESC;

Quais jogadores têm a maior quantidade de impedimentos no campeonato (+ ano)?

SELECT
    cbg.atleta,
    date_part('year', cbf.data) AS ano,
    SUM(cbe.impedimentos) AS total_impedimentos
FROM brasileiro_pontos_corridos.campeonato_brasileiro_gols cbg
JOIN brasileiro_pontos_corridos.campeonato_brasileiro_full cbf ON cbg.partida_id = cbf.id
JOIN brasileiro_pontos_corridos.campeonato_brasileiro_estatisticas_full cbe ON cbf.id = cbe.partida_id AND cbg.clube = cbe.clube
GROUP BY cbg.atleta, date_part('year', cbf.data)
ORDER BY total_impedimentos DESC;
